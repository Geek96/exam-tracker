# p9 Demo 引导流程 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a "线性代数 Demo" course that auto-launches a 10-step guided side-panel tour when entered, walking new users through all key app features in zh/en/es.

**Architecture:** A new `demo-tour.js` module is loaded in `course.html` and activated by `course.js` only when `courseId === '__demo__'`. It seeds pre-built chapters and a Markdown material into IndexedDB, renders a fixed 220 px right-side panel, advances through 10 steps as the user interacts, and exposes a reset function that clears demo data and restarts.

**Tech Stack:** Vanilla JS (no ES modules), IndexedDB (reuses `getDB`, `getChatDB`, `getChunkDB` from course.js), localStorage, existing `strings.js` i18n (`window.t` / `window.tf`).

  };
  existing.push(demo);

}

```js
async function dbDeleteChunksForFile(fileId) {
  const db = await getChunkDB();
  return new Promise((res, rej) => {
    const tx = db.transaction('materialChunks', 'readwrite');
    const store = tx.objectStore('materialChunks');
    const req = store.getAll();
    req.onsuccess = () => {
      req.result.filter(c => c.fileId === fileId).forEach(c => store.delete(c.id));
    };
    tx.oncomplete = res;
    tx.onerror = () => rej(tx.error);
  });
}
```

  // ── i18n shortcuts ──────────────────────────────────────────────────────────
  function t(key)       { return (window.t  && window.t(key))       || key; }
  function tf(key, vars){ return (window.tf && window.tf(key, vars)) || key; }

  // ── State ───────────────────────────────────────────────────────────────────
  var stepIndex     = 0;
  var questionsSent = 0;

  // ── Panel DOM refs (bound after HTML is parsed) ─────────────────────────────
  var panel, stepCircle, stepLabel, progressFill, titleEl, descEl, extraEl, skipBtn, nextBtn, resetBtn;

  function bindRefs() {
    panel        = document.getElementById('demoTourPanel');
    stepCircle   = document.getElementById('dtpStepCircle');
    stepLabel    = document.getElementById('dtpStepLabel');
    progressFill = document.getElementById('dtpProgressFill');
    titleEl      = document.getElementById('dtpTitle');
    descEl       = document.getElementById('dtpDesc');
    extraEl      = document.getElementById('dtpExtra');
    skipBtn      = document.getElementById('dtpSkipBtn');
    nextBtn      = document.getElementById('dtpNextBtn');
    resetBtn     = document.getElementById('dtpResetBtn');
  }

  // ── Step definitions ────────────────────────────────────────────────────────
  // autoOn values: null | 'examModalOpened' | 'examAdded' | 'chapterChecked'
  //               | 'fileSelected' | 'questionSent' | 'materialSaved'
  function getSteps() {
    return [
      { target: '#courseHero',           title: t('demoWelcomeTitle'),    desc: t('demoWelcomeDesc'),    showNext: true,  autoOn: null },
      { target: '#btnAddExamCourse',     title: t('demoAddExamTitle'),    desc: t('demoAddExamDesc'),    showNext: false, autoOn: 'examModalOpened' },
      { target: '#courseExamModalOverlay', title: t('demoConfirmExamTitle'), desc: t('demoConfirmExamDesc'), showNext: false, autoOn: 'examAdded' },
      { target: '#chapterNavCard',       title: t('demoChaptersTitle'),   desc: t('demoChaptersDesc'),   showNext: true,  autoOn: null },
      { target: '#chaptersList',         title: t('demoCheckChapterTitle'),desc: t('demoCheckChapterDesc'),showNext: false, autoOn: 'chapterChecked' },
      { target: '#panelOverview',        title: t('demoMaterialsTitle'),  desc: t('demoMaterialsDesc'),  showNext: true,  autoOn: null },
      { target: '.panel-right',          title: t('demoAITitle'),         desc: t('demoAIDesc'),         showNext: true,  autoOn: null },
      { target: '#aiContextBadge',       title: t('demoFileSelectTitle'), desc: t('demoFileSelectDesc'), showNext: false, autoOn: 'fileSelected' },
      { target: '#aiChatInput',          title: t('demoQuestionsTitle'),  desc: t('demoQuestionsDesc'),  showNext: false, autoOn: 'questionSent', extra: 'questions' },
      { target: '.btn-save-msg',         title: t('demoSaveTitle'),       desc: t('demoSaveDesc'),       showNext: false, autoOn: 'materialSaved' },
    ];
  }

  // ── Highlight ────────────────────────────────────────────────────────────────
  var _highlighted = null;
  function highlight(sel) {
    if (_highlighted) { _highlighted.classList.remove('demo-tour-highlight'); }
    _highlighted = sel ? document.querySelector(sel) : null;
    if (_highlighted) _highlighted.classList.add('demo-tour-highlight');
  }
  function clearHighlight() {
    if (_highlighted) { _highlighted.classList.remove('demo-tour-highlight'); _highlighted = null; }
  }

  // ── Render step ─────────────────────────────────────────────────────────────
  function renderStep(idx) {
    var steps = getSteps();
    if (idx >= steps.length) { showCompletion(); return; }
    var step = steps[idx];
    var n = idx + 1;
    stepCircle.textContent = n;
    stepLabel.textContent  = tf('demoStepOf', { n: n });
    progressFill.style.width = Math.round((n / TOTAL_STEPS) * 100) + '%';
    titleEl.textContent = step.title;
    descEl.textContent  = step.desc;
    extraEl.innerHTML   = '';

    skipBtn.textContent       = t('demoSkip');
    skipBtn.style.display     = '';
    nextBtn.textContent       = t('demoNext');
    nextBtn.style.display     = step.showNext ? '' : 'none';
    resetBtn.textContent      = t('demoReset');

    highlight(step.target);
    if (step.extra === 'questions') renderQuestionButtons();
    setupAutoAdvance(step);
  }

  function showCompletion() {
    clearHighlight();
    clearAutoListeners();
    stepCircle.textContent    = '✓';
    stepLabel.textContent     = '';
    progressFill.style.width  = '100%';
    titleEl.textContent       = t('demoDoneTitle');
    descEl.textContent        = t('demoDoneDesc');
    extraEl.innerHTML         = '';
    skipBtn.style.display     = 'none';
    nextBtn.style.display     = 'none';
    resetBtn.textContent      = t('demoReset');
    localStorage.setItem('demoTourDismissed', '1');
  }

  // ── Question buttons (step 9) ────────────────────────────────────────────────
  function renderQuestionButtons() {
    questionsSent = 0;
    var qs = [t('demoQ1'), t('demoQ2'), t('demoQ3')];
    extraEl.innerHTML = '';
    qs.forEach(function (q) {
      var btn = document.createElement('button');
      btn.className = 'dtp-q-btn';
      btn.textContent = q;
      btn.addEventListener('click', function () {
        if (btn.classList.contains('sent')) return;
        btn.classList.add('sent');
        var inputEl  = document.getElementById('aiChatInput');
        var sendBtn  = document.getElementById('aiChatSend');
        if (inputEl && sendBtn) {
          inputEl.value = q;
          inputEl.dispatchEvent(new Event('input'));
          sendBtn.click();
        }
        questionsSent++;
        if (questionsSent >= 1) nextBtn.style.display = '';
      });
      extraEl.appendChild(btn);
    });
  }

  // ── Auto-advance listener management ────────────────────────────────────────
  var _cleanups = [];
  function clearAutoListeners() {
    _cleanups.forEach(function (fn) { try { fn(); } catch(e){} });
    _cleanups = [];
  }

  function setupAutoAdvance(step) {
    clearAutoListeners();

    if (step.autoOn === 'examModalOpened') {
      var overlay = document.getElementById('courseExamModalOverlay');
      if (overlay) {
        var obs = new MutationObserver(function () {
          if (overlay.classList.contains('open')) { advance(); }
        });
        obs.observe(overlay, { attributes: true, attributeFilter: ['class'] });
        _cleanups.push(function () { obs.disconnect(); });
      }
    }

    if (step.autoOn === 'examAdded') {
      var form = document.getElementById('courseExamForm');
      if (form) {
        var fn = function () { setTimeout(advance, 300); };
        form.addEventListener('submit', fn);
        _cleanups.push(function () { form.removeEventListener('submit', fn); });
      }
    }

    if (step.autoOn === 'chapterChecked') {
      var list = document.getElementById('chaptersList');
      if (list) {
        var fn2 = function (e) {
          if (e.target.classList.contains('chapter-check') || e.target.classList.contains('section-check')) {
            setTimeout(advance, 350);
          }
        };
        list.addEventListener('change', fn2);
        _cleanups.push(function () { list.removeEventListener('change', fn2); });
      }
    }

    if (step.autoOn === 'fileSelected') {
      var pickerList = document.getElementById('aiFilePickerList');
      if (pickerList) {
        var fn3 = function (e) {
          if (e.target.closest('.ai-file-picker-row')) { setTimeout(advance, 400); }
        };
        pickerList.addEventListener('click', fn3);
        _cleanups.push(function () { pickerList.removeEventListener('click', fn3); });
      }
    }

    if (step.autoOn === 'materialSaved') {
      var msgs = document.getElementById('aiChatMessages');
      if (msgs) {
        var obs2 = new MutationObserver(function () {
          // Wire click on newly appeared save buttons (mark with class to avoid duplicate wiring)
          msgs.querySelectorAll('.btn-save-msg:not(.demo-wired)').forEach(function (saveBtn) {
            saveBtn.classList.add('demo-wired');
            clearHighlight();
            highlight('.btn-save-msg');
            saveBtn.addEventListener('click', function () {
              setTimeout(showCompletion, 800);
            });
          });
        });
        obs2.observe(msgs, { childList: true, subtree: true });
        _cleanups.push(function () { obs2.disconnect(); });
      }
    }
  }

  // ── Advance / dismiss ────────────────────────────────────────────────────────
  function advance() {
    clearAutoListeners();
    stepIndex++;
    renderStep(stepIndex);
  }

  function dismiss() {
    clearHighlight();
    clearAutoListeners();
    panel.style.display = 'none';
    document.body.classList.remove('demo-tour-active');
    localStorage.setItem('demoTourDismissed', '1');
    var banner = document.getElementById('demoRestartBanner');
    var restartBtn = document.getElementById('demoRestartBtn');
    if (banner && restartBtn) {
      restartBtn.textContent = t('demoRestart');
      banner.style.display = '';
    }
  }

  // ── Start tour ───────────────────────────────────────────────────────────────
  function startTour() {
    stepIndex = 0; questionsSent = 0;
    panel.style.display = 'flex';
    document.body.classList.add('demo-tour-active');
    document.getElementById('demoRestartBanner').style.display = 'none';
    localStorage.removeItem('demoTourDismissed');
    renderStep(0);
  }

  // ── Wire panel buttons ───────────────────────────────────────────────────────
  function wirePanelButtons() {
    skipBtn.addEventListener('click', function () {
      if (stepIndex < TOTAL_STEPS - 1) advance(); else dismiss();
    });
    nextBtn.addEventListener('click', advance);
    resetBtn.addEventListener('click', function () {
      if (confirm(t('demoResetConfirm'))) resetDemo();
    });
    var restartBtn = document.getElementById('demoRestartBtn');
    if (restartBtn) {
      restartBtn.addEventListener('click', function () { startTour(); });
    }
  }

  // ── Pre-fill exam modal when Add Exam is clicked ─────────────────────────────
  function patchExamModalOpen() {
    var addBtn = document.getElementById('btnAddExamCourse');
    if (!addBtn) return;
    addBtn.addEventListener('click', function () {
      setTimeout(function () {
        var nameInput = document.getElementById('courseExamInputName');
        var dateInput = document.getElementById('courseExamInputDate');
        var lang = localStorage.getItem('app_lang') || 'zh';
        var examNames = { zh: '期末考试', en: 'Final Exam', es: 'Examen Final' };
        if (nameInput && !nameInput.value) nameInput.value = examNames[lang] || examNames.zh;
        if (dateInput && !dateInput.value) {
          var d = new Date(); d.setDate(d.getDate() + 30);
          dateInput.value = d.toISOString().split('T')[0];
        }
        document.querySelectorAll('#courseExamTypeGroup .exam-type-btn').forEach(function (btn) {
          btn.classList.toggle('selected', btn.dataset.type === 'final');
        });
      }, 60);
    });
  }

  // ── Seed chapters and Markdown material ──────────────────────────────────────
  async function seedDemoData() {
    // Seed chapters if course has none
    if (course && course.chapters && course.chapters.length === 0) {
      _setCourseChapters(getDemoChapters());
    }
    // Seed Markdown material into IndexedDB if not already present
    try {
      var all = await dbGetAll();
      var hasMd = all.some(function (f) { return f.id === DEMO_MD_FILE_ID; });
      if (!hasMd) {
        var enc = new TextEncoder().encode(DEMO_MD).buffer;
        var record = {
          id: DEMO_MD_FILE_ID,
          courseId: DEMO_ID,
          name: DEMO_MD_NAME,
          type: 'text/markdown',
          size: enc.byteLength,
          data: enc,
          addedAt: new Date().toISOString(),
        };
        await dbSave(record);
        await ensureChunksForMaterial(record);
        await renderMaterials();
      }
    } catch (e) {
      console.warn('[DemoTour] seedDemoData failed', e);
    }
  }

  // ── Reset demo ───────────────────────────────────────────────────────────────
  async function resetDemo() {
    // 1. Clear exams
    try {
      var allExams = JSON.parse(localStorage.getItem('examTrackerExams') || '[]');
      var kept = allExams.filter(function (e) { return e.courseId !== DEMO_ID; });
      localStorage.setItem('examTrackerExams', JSON.stringify(kept));
      _reloadCourseExams();
    } catch (e) { console.warn('[DemoTour] clear exams failed', e); }

    // 2. Reset chapters
    _setCourseChapters(getDemoChapters());

    // 3. Clear chat sessions
    try { await dbSaveSessions(DEMO_ID, []); } catch (e) { console.warn('[DemoTour] clear sessions failed', e); }
    if (typeof initSessions === 'function') await initSessions().catch(function(){});

    // 4. Remove user-uploaded materials, keep seeded MD
    try {
      var files = await dbGetAll();
      for (var i = 0; i < files.length; i++) {
        var f = files[i];
        if (f.courseId === DEMO_ID && f.id !== DEMO_MD_FILE_ID) {
          await dbDelete(f.id);
          await dbDeleteChunksForFile(f.id);
        }
      }
      await renderMaterials();
    } catch (e) { console.warn('[DemoTour] clear materials failed', e); }

    // 5. Clear AI file selection
    try { localStorage.removeItem('aiFileSelection_' + DEMO_ID); } catch (e) {}

    // 6. Restart tour
    startTour();
  }

  // ── initDemoTour (called by course.js) ──────────────────────────────────────
  async function initDemoTour() {
    bindRefs();
    if (!panel) return;

    await seedDemoData();
    patchExamModalOpen();
    wirePanelButtons();

    if (localStorage.getItem('demoTourDismissed') === '1') {
      var restartBtn = document.getElementById('demoRestartBtn');
      if (restartBtn) {
        restartBtn.textContent = t('demoRestart');
        document.getElementById('demoRestartBanner').style.display = '';
      }
    } else {
      startTour();
    }
  }

  window.initDemoTour = initDemoTour;
})();
